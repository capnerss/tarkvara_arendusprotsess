Ülesanne: Valguskett

Töö kirjeldus:

Luua valguskett, mis koosneb mitmest valgusdioodist ja töötab mitmes erinevas valgustusrežiimis, mida juhitakse potentsiomeetriga.

  1. Kõik valgusdioodid vilguvad üheaegselt.
  2. Kõik valgusdioodid vilguvad kordamööda.
  3. Valgusdioodid vilguvad rühmadena.
  4. Valgusdioodid vilguvad juhuslikus järjekorras (küünla efekt).
  5. Jooksev tuli edasi-tagasi.

Vaja läheb:

  • Potentsiomeeter - 1 tk
  • LED-lamp - 5 tk
  • Takisti (220 Oomi) - 5 tk
  • Juhe - 10 tk
  • Lisatoide 5V

Ühendamise skeem:

Simulatsioon:

Keera liugurit (potentsiomeetrit), et vahetada töörežiime (1 kuni 6).

Valgusdioodid

Režiim: 1 (Välja lülitatud)

AnalogRead: 0

Programmi kood:

// Globaalne massiiv viikude mugavaks haldamiseks
int ledPins[] = {13, 12, 11, 10, 9};
int sensorPin = A0;

void setup() {
  for (int i = 0; i < 5; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
  Serial.begin(9600);
  randomSeed(analogRead(A1));
}

// 1. Kõik LED-id vilguvad korraga
void mode1() {
  for (int i = 0; i < 5; i++) {
    for (int j = 0; j < 5; j++) digitalWrite(ledPins[j], HIGH);
    delay(200);
    for (int j = 0; j < 5; j++) digitalWrite(ledPins[j], LOW);
    delay(200);
  }
}

// 2. LED-id vilguvad kordamööda
void mode2() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPins[i], HIGH);
    delay(200);
    digitalWrite(ledPins[i], LOW);
  }
}
// 3. "Värviline" režiim (grupeerimine)
void mode3() {
  digitalWrite(ledPins[0], HIGH); digitalWrite(ledPins[4], HIGH);
  delay(200);
  digitalWrite(ledPins[0], LOW); digitalWrite(ledPins[4], LOW);
  delay(500);

  digitalWrite(ledPins[1], HIGH); digitalWrite(ledPins[3], HIGH);
  delay(200);
  digitalWrite(ledPins[1], LOW); digitalWrite(ledPins[3], LOW);
  delay(500);

  digitalWrite(ledPins[2], HIGH);
  delay(200);
  digitalWrite(ledPins[2], LOW);
}

// 4. Juhuslik järjekord (küünla efekt)
void mode4() {
  for (int i = 0; i < 10; i++) {
    int randPin = ledPins[random(0, 5)];
    digitalWrite(randPin, HIGH);
    delay(random(100, 300));
    digitalWrite(randPin, LOW);
  }
}

// 5. Kõik töötavad (jooksev tuli edasi-tagasi)
void mode5() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPins[i], HIGH);
    delay(150);
    digitalWrite(ledPins[i], LOW);
  }
  for (int i = 3; i > 0; i--) {
    digitalWrite(ledPins[i], HIGH);
    delay(150);
    digitalWrite(ledPins[i], LOW);
  }
}

// Kõikide dioodide väljalülitamine
void turnOffAll() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPins[i], LOW);
  }
}

void loop() {
  int sensorValue = analogRead(sensorPin);

  // Jagame 7 osaks võrdsete intervallide saamiseks
  int mode = map(sensorValue, 0, 1023, 1, 7);
  if (mode > 6) mode = 6;

  if (mode == 2) {
    mode1();
  }
  else if (mode == 3) {
    mode2();
  }
  else if (mode == 4) {
    mode3();
  }
  else if (mode == 5) {
    mode4();
  }
  else if (mode == 6) {
    mode5();
  }
  else {
    turnOffAll();
  }
}
