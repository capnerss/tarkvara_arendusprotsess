Задание: Световая цепочка

Описание работы:

Создать световую цепочку, состоящую из нескольких светодиодов, работающих в нескольких различных режимах освещения, управляемых потенциометром.

  1. Все светодиоды мигают одновременно.
  2. Все светодиоды мигают по очереди.
  3. Светодиоды мигают группами.
  4. Светодиоды мигают в случайном порядке (эффект свечи).
  5. Бегущий огонек туда и обратно.

Понадобится:

  • Потенциометр - 1 шт.
  • LED-лампа - 5 шт.
  • Резистор (220 Ом) - 5 шт.
  • Провода - 10 шт.
  • Дополнительное питание 5V

Симуляция:

Покрутите ползунок (потенциометр), чтобы переключать режимы работы (от 1 до 6).

Светодиоды

Режим: 1 (Выключено)

AnalogRead: 0

Программный код (Arduino C++):

// Глобальный массив пинов для удобного управления
int ledPins[] = {13, 12, 11, 10, 9};
int sensorPin = A0;

void setup() {
  for (int i = 0; i < 5; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
  Serial.begin(9600);
  randomSeed(analogRead(A1));
}

// 1. Все светодиоды мигают одновременно
void mode1() {
  for (int i = 0; i < 5; i++) {
    for (int j = 0; j < 5; j++) digitalWrite(ledPins[j], HIGH);
    delay(200);
    for (int j = 0; j < 5; j++) digitalWrite(ledPins[j], LOW);
    delay(200);
  }
}

// 2. Светодиоды мигают по очереди
void mode2() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPins[i], HIGH);
    delay(200);
    digitalWrite(ledPins[i], LOW);
  }
}

// 3. "Цветной" режим (группировка)
void mode3() {
  digitalWrite(ledPins[0], HIGH); digitalWrite(ledPins[3], HIGH);
  delay(200);
  digitalWrite(ledPins[0], LOW); digitalWrite(ledPins[3], LOW);

  digitalWrite(ledPins[1], HIGH);
  delay(200);
  digitalWrite(ledPins[1], LOW);

  digitalWrite(ledPins[2], HIGH); digitalWrite(ledPins[4], HIGH);
  delay(200);
  digitalWrite(ledPins[2], LOW); digitalWrite(ledPins[4], LOW);
}

// 4. Случайный порядок (эффект свечи)
void mode4() {
  for (int i = 0; i < 10; i++) {
    int randPin = ledPins[random(0, 5)];
    digitalWrite(randPin, HIGH);
    delay(random(100, 300));
    digitalWrite(randPin, LOW);
  }
}

// 5. Работают все (бегущий огонек туда-обратно)
void mode5() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPins[i], HIGH);
    delay(150);
    digitalWrite(ledPins[i], LOW);
  }
  for (int i = 3; i > 0; i--) {
    digitalWrite(ledPins[i], HIGH);
    delay(150);
    digitalWrite(ledPins[i], LOW);
  }
}

// Выключение всех диодов
void turnOffAll() {
  for (int i = 0; i < 5; i++) {
    digitalWrite(ledPins[i], LOW);
  }
}

void loop() {
  int sensorValue = analogRead(sensorPin);

  // Разбиваем на 7 частей для равных интервалов
  int mode = map(sensorValue, 0, 1023, 1, 7);
  if (mode > 6) mode = 6;

  if (mode == 2) {
    mode1();
  }
  else if (mode == 3) {
    mode2();
  }
  else if (mode == 4) {
    mode3();
  }
  else if (mode == 5) {
    mode4();
  }
  else if (mode == 6) {
    mode5();
  }
  else {
    turnOffAll();
  }
}
